  \subsection{XML et XSL}
    \begin{frame}
      \frametitle{XML (1)}
      \framesubtitle{Historique et Objectifs}
      \begin{itemize}
        \item Définie par le W3C (première TR en 1998).
        \item Né de la fusion du meilleur de SGML et de HTML :
          \begin{itemize}
            \item la polyvalence de SGML,
            \item la simplicité de HTML.
          \end{itemize}
        \item Trois principaux objectifs :
          \begin{itemize}
            \item l'écriture de programmes de traitements doit-être aisé,
            \item la définition de XML doit-être formelle et concise,
            \item La verbosité des document est peu importante.
          \end{itemize}   
        \begin{example}
          Example de doc XML
        \end{example}
      \end{itemize}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML (2)}
      \framesubtitle{Structure et Syntaxe}
      \begin{itemize}
        \item Langage à balise à inclusion strict.
        \item Le nombre de concepts est limité :
          \begin{itemize}
            \item balises,
            \item attributs,
            \item données.
          \end{itemize}
      \end{itemize}
    \end{frame}

  \subsection{XML et XSL}
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Historique et Objectifs}
      \begin{itemize}
        \item Définie par le W3C (première TR en 1998).
        \item Né de la fusion du meilleur de SGML et de HTML :
          \begin{itemize}
            \item la polyvalence de SGML,
            \item la simplicité de HTML.
          \end{itemize}
        \item 4 principaux objectifs :
          \begin{itemize}
            \item L'interopérabilité doit-être assurée,
            \item l'écriture de programmes de traitements doit-être aisée,
            \item la définition de XML doit-être formelle et concise,
            \item La verbosité des documents n'est pas un problème.
          \end{itemize}   
      \end{itemize}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Structure et Syntaxe}
      \begin{itemize}
        \item Langage à balises à inclusion stricte structurant de manière arborescente les documents.
        \item Le nombre de concepts est limité :
          \begin{itemize}
            \item balises,
            \item attributs,
            \item données.
          \end{itemize}
      \end{itemize}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Exemple}
       \begin{Example}
       (...) \newline
       <annuaire> \newline
        <personne catégorie="jedi"> \newline
          <nom>Luke Skywalker</nom> \newline
          <email>luke@sw.com</email> \newline
        </persone> \newline
        ... \newline
       </annuaire> \newline     
       (...)
       \end{Example}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Modèle de documents}
      \begin{itemize}
        \item Définition de modèles de documents :
          \begin{itemize}
            \item DTD, Document Type Definition,
            \item XML Schema...
          \end{itemize}
        \item 2 niveaux de validité :
          \begin{itemize}
            \item document bien formé,
            \item document valide.
          \end{itemize}
        \item Un modèle définit un dialecte XML.
        \item Exemple de dialectes : \emph SVG, MathML, SOAP, XHTML...
      \end{itemize}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Outils de traitements}
      \begin{itemize}
        \item DOM, Document Object Model
          \begin{itemize}
            \item Défini par le W3C.
            \item Se base sur la structure arborescente de XML.
            \item Empreinte mémoire volumineuse.
          \end{itemize}
        \item SAX, Simple API for XML
          \begin{itemize}
            \item Né d'une discussion entre utilisateurs de XML et de Java.
            \item La lecture d'un noeud génère un évènement à récupérer.
            \item Empreinte mémoire faible, mais accès en lecture seule.
          \end{itemize}  
      \end{itemize}
    \end{frame}
    
    \begin{frame}
      \frametitle{XML}
      \framesubtitle{Transformation de documents XML}
      \begin{center}
       \includegraphics[scale=0.10]{../books/bm-files/images/TransformationXML.png}
      \end{center}
      \begin{itemize}
        \item 4 classes de transformations XML :
        \begin{itemize}
          \item en bloc, comme XSLT,
          \item en "streaming", 
          \item incrémentales,
          \item inversibles.
        \end{itemize}
      \end{itemize}
    \end{frame}

\begin{frame}
  \frametitle{XSL}
  \framesubtitle{Introduction}
  \begin{itemize}
    \item XSL est un groupe de 3 standards :
    \begin{itemize}
      \item XPath, langage de navigation dans un document XML,
      \item XSLT, langage de transformation,
      \item XSL-FO, dialecte XML de mise en forme (non détaillé ici).
    \end{itemize}
  \end{itemize}
\end{frame}

\begin{frame}
  \frametitle{XSL}
  \framesubtitle{XSLT, Introduction}
  \begin{itemize}
    \item Langage déclaratif sans effet de bord.
    \item Notion d'arbre source et résultat.
    \item Une règle de transformation est nommée "template".
    \item Se base sur XPath pour identifier les sous-ensembles de l'arbre source.
  \end{itemize}
\end{frame} 

\begin{frame}
  \frametitle{XSL}
  \framesubtitle{XSLT, Algorithme général}
  \begin{itemize}
    \item Lecture de la feuille XSLT et du document source.
    \item Création d'un arbre de transformation, un arbre source et un arbre résultat (vide).
    \item Traitement du noeud racine puis par propagation, des autres noeuds (processus d'instanciation).
    \item Linéarisation de l'arbre résultat.
  \end{itemize}  
\end{frame}

  \subsection{Traitements Incrémentaux}
  \begin{frame}
    \frametitle{Traitements Incrémentaux}
    \framesubtitle{Principes}
    \begin{itemize}
    \item « Étant donnés un programme f et une opération OP, un programme f' est appelé une version incrémentale de f par rapport à OP si f' calcule f (x OP delta(x)) de façon efficace en utilisant f (x).»
    \end{itemize}
\begin{center}
     \includegraphics[scale=0.07]{../books/bm-files/images/calculIncr2.png}
\end{center}
  \end{frame}

  \begin{frame}
    \frametitle{Traitements Incrémentaux}
    \framesubtitle{Méthodes générales}
    \begin{itemize}
    \item<1-> Méthode de recalcul sélectif
      \begin{itemize}
        \item Technique de memoizing
        \item Problème : stockage des résultats intermédiaires dans le cas d'un langage avec effets de bords.
      \end{itemize}
    \item<2-> Méthode par différence finie
      \begin{itemize}
        \item Hypothèse d'une fonction de différence.
      \end{itemize}
    \item<3-> Méthode d'évaluation partielle
      \begin{itemize}
        \item Partitionnement du domaine "d'entrée".
        \item Difficulté : trouver un bon partionnement.
      \end{itemize}      
    \end{itemize}
  \end{frame}

  \begin{frame}
    \frametitle{Traitements Incrémentaux}
    \framesubtitle{Méthodes spécifiques au génie documentaire}
    \begin{itemize}
    \item<1-> Traduction incrémentale
      \begin{itemize}
        \item Traduction d'arbre (algorithme de Lindén). 
        \item Repose sur les grammaires d'entrée et de sortie.
      \end{itemize}
    \item<2-> Traitement au plus tard
      \begin{itemize}
        \item retarde au maximum le traitement car :
        \begin{itemize}
        \item certains sous-calculs peuvent ne pas être pertinents,
        \item non nécessaires immédiatement, 
        \item ou bien les résultats redondants.
        \end{itemize}
      \end{itemize}    
    \item<3-> Méthode ad-hoc
    \begin{itemize}
    \item<3-> pour XSLT: incXslt \newline Détaillé ci-après.
    \end{itemize}
    \end{itemize}
  \end{frame}
  
  \begin{frame}
    \frametitle{Traitements Incrémentaux}
    \framesubtitle{incXSLT, Principes}
    \begin{itemize}
      \item Création d'un graphe de dépendances entre les templates,
      \item Construction de règles de ré-évaluation construites à partir des expressions XPath,
      \item Analyse de l'arbre d'exécution afin de restaurer les contextes.
    \end{itemize}
  \end{frame}
  
  %% Mika
  \subsection{Transformations Inverses}
  \begin{frame}
    \frametitle{Transformations Inverses}
    \framesubtitle{Principes}
    \begin{itemize}
      \item Répercuter les modifications de l'arbre cible dans l'arbre source.
      \item Soit f et f' resp. une transformation et son inverse,
      \item et S et T resp. les arbres source et cible tel que f(S) = T et f'(T)=S.
      \item On souhaite que f(f'(T)) = T et f'(f(S)) = S
      \item Reviens à une détection d'isomorphisme.
    \end{itemize}
  \end{frame}

  \begin{frame}
    \frametitle{Transformations Inverses}
    \framesubtitle{Solutions possibles via XSLT}
    \begin{itemize}
      \item Détecter les sous arbres de la transformation qui sont des isomorphismes.
      \item Classifier les éléments XSLT de production.
      \item Garder les liens entre les noeuds générés, les noeuds sources et les noeuds de productions.
      \item Classifier les types de modifications possibles dans l'arbre résultat.
    \end{itemize}
  \end{frame}
